Basic Examples

rng(123);
% Imaginary array
ar_img = rand([1,7]) + 1i*rand([1,7]);
% Regular Array
ar_real = rand([1,10]);
% Combine arrays
ar_full = [ar_real ar_img];
ar_full = ar_full(randperm(length(ar_full)));
disp(ar_full);
